var str = "xyz";
str.search(/x/); // 0
str.search(/y/); // 1
str.search(/w/); // -1
str.match(/y/);
// ['y', index: 1, input: 'xyz']

str.match(/w/); // null

str.match(/../);
// ['xy', index: 0, input: 'xyz']

str.search(/.(.)/);
// ['xy', 'y', index: 0, input: 'xyz']

str.search(/(.)(.)/);
// ['xy', 'x', 'y', index: 0, input: 'xyz']
